163 // Rotate all of the cells using the rotation matrix
164 for (int i = 0; i < cells.Length; i++)

169

175 cell.x -= 0.5f;
176 cell.y -= 0.5f;
177 x = Mathf.CeilToInt((cell.x * matrix[0] * direction) + (cell.y * matrix[1] * direction));
178 y = Mathf.CeilToInt((cell.x * matrix[2] * direction) + (cell.y * matrix[3] * direction));
179 break;
180
181 default:
182 x = Mathf.RoundToInt((cell.x * matrix[0] * direction) + (cell.y * matrix[1] * direction));
183 y = Mathf.RoundToInt((cell.x * matrix[2] * direction) + (cell.y * matrix[3] * direction));
184 break;
